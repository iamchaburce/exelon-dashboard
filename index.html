" [" + item.catalogNo + "]" : ""), data.approvalRequired, // Approval Required? "", // Approvers (auto from Setup) "Pending", // Approval Status "⬜ Pending", // Status "", // Linked BOM Tab "", // PO # (blank until issued) data.remarks // Remarks ]); }); return ContentService.createTextOutput("OK"); }
